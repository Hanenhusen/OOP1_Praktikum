/**
* Ermittelt zu einer Karte, die an der angegebenen Position liegt, die Nachbarkarte in der angegebenen Richtung.
* Wenn die Nachbar-Position außerhalb des Spielfeld liegt, oder wenn für die richtung ein
* ungültiger Wert angegeben wird, wird null zurückgegeben.
* @param zeile Position der Karte
* @param spalte Position der Karte
* @param richtung - erlaubte Werte sind "oben", "unten", "rechts", "links"
* @return nachbar Nachbar-Karte, die in der angegebenen Richtung neben der Karte liegt, null im Fehlerfall
*/
Karte nachbarKarteInRichtung(int zeile, int spalte, String richtung) {

  // Fallunterscheidung nach richtung
  switch(richtung) {
      case "links" : spalte -=1; break;
      case "rechts": spalte +=1; break;
      case "oben"  : zeile  -=1; break;
      case "unten" : zeile  +=1; break;
    default: return null;    
    }

  // unerlaubte Positions-Angaben führen zur Rückgabe von null
  if(zeile <0 || zeile>= feld.length)
      return null;
    if(spalte <0 || spalte >=feld[0].length)
      return null;

  // jetzt sind alle Prüfungen ok, die Karte an der Stelle kann zurückgegeben werden.
  return feld[zeile][spalte];
}
